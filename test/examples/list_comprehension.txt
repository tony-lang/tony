% := (a, b) => eval('{a} % {b}')
== := (a, b) => eval('{a} == {b}')
< := (a, b) => eval('{a} < {b}')

==================
list comprehension
==================

print([x + 1 | x in [1, 2, 3]])

---

==================
conditionals
==================

print([x + 1 | x in [1, 2, 3] if x < 3])

---

[ 2, 3 ]

==================
multi dimension
==================

print([x + y | x in [1, 2, 3], y in [4, 5] if x + y % 2 == 0])

---

[ 6, 6, 8 ]
